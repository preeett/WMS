#include #include #include #define SCREEN_WIDTH 128 // OLED display width, in pixels #define SCREEN_HEIGHT 64 // OLED display height, in pixels Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1); // Function to draw more snowflakes void drawSnowflakes() { for (int i = 0; i < 20; i++) { // Add 20 snowflakes int x = random(0, display.width()); int y = random(0, display.height()); display.drawPixel(x, y, WHITE); } } // Function to center-align the text void displayCenteredText(String text, int yPos) { int textWidth = display.strWidth(text.c_str()); int centerX = (display.width() - textWidth) / 2; display.setCursor(centerX, yPos); display.print(text); } // Function to right-align the text void displayRightAlignedText(String text, int yPos) { int textWidth = display.strWidth(text.c_str()); int rightX = display.width() - textWidth; display.setCursor(rightX, yPos); display.print(text); } // Function to display data with alignment void displayData() { display.clearDisplay(); // Clear the display first // Draw snowflakes drawSnowflakes(); // Display temperature (center aligned) String temp = "Temp: 24C"; displayCenteredText(temp, 10); // Display humidity (right aligned) String humidity = "Humidity: 60%"; displayRightAlignedText(humidity, 30); // Display air pressure (left aligned as an example) display.setCursor(0, 50); display.print("Pressure: 1012 hPa"); // Show everything on the display display.display(); } void setup() { // Initialize the OLED display if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) { Serial.println(F("SSD1306 allocation failed")); for (;;); // Don't proceed, loop forever } display.clearDisplay(); // Clear the display buffer } void loop() { displayData(); // Call function to display the data delay(2000); // Delay before refreshing the screen }
